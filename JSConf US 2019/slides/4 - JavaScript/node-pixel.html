const pixel = require('node-pixel');
const five = require('johnny-five');
const rvl = require('rvl-node-animations');
const board = new five.Board();

board.on('ready', async () => {
  await rvl.initRenderer();
  strip = new pixel.Strip(options)
  strip.on('ready', () => {
    const layer = rvl.createRainbowWave(255, 4);
    const parameters = rvl.createWaveParameters(layer);
    const pixels = rvl.renderPixels(parameters);
    for (let i = 0; i < pixels.length; i++) {
      strip.pixel(i).color([ pixels[i].r, pixels[i].g, pixels[i].b ]);
    }
  });
});
